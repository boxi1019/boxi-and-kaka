<script>
const KEY = "lottery_activity_data";
let allDocs = [], currentDocId = "";
let currentAngle = 0, isSpinning = false, selectedPrize = null, currentNick = "";
let myPrizeList = [], tradeData = [], prizes = [], nickNameList = [];

const step1 = document.getElementById("step1");
const step2 = document.getElementById("step2");
const activitySelect = document.getElementById("activitySelect");
const goStep2 = document.getElementById("goStep2");
const pageTitle = document.getElementById("pageTitle");
const timeTip = document.getElementById("timeTip");
const wheel = document.getElementById("wheel");
const spinBtn = document.getElementById("spinBtn");
const result = document.getElementById("result");
const myPrizes = document.getElementById("myPrizes");
const wantText = document.getElementById("wantText");
const publishTrade = document.getElementById("publishTrade");
const tradeList = document.getElementById("tradeList");
const emptyTip = document.getElementById("emptyTip");
const nickSelect = document.getElementById("nickSelect");
const nickTip = document.getElementById("nickTip");

function loadFromLocal(){
  const s = localStorage.getItem(KEY);
  allDocs = s ? JSON.parse(s) : [];
}

function renderActivitySelect(){
  let html = '<option value="">-- 选择活动 --</option>';
  allDocs.forEach(d=>html += `<option value="${d.id}">${d.name}</option>`);
  activitySelect.innerHTML = html;
}

function checkActivityTime(startTimeStr){
  if(!startTimeStr) return true;
  const now = new Date().getTime();
  const start = new Date(startTimeStr).getTime();
  return now >= start;
}

function loadActivity(docId){
  currentDocId = docId;
  const doc = allDocs.find(d=>d.id===docId);
  if(!doc) return;

  pageTitle.innerText = doc.title;
  nickNameList = doc.nicks;
  prizes = doc.prizes;
  myPrizeList = doc.myPrizeList;
  tradeData = doc.tradeData;
  selectedPrize = null;
  currentNick = "";

  const canStart = checkActivityTime(doc.startTime);
  if(doc.startTime){
    timeTip.innerText = canStart ? "✅ 活动已开始，可以抽奖" : `⏰ 活动未开始，开始时间：${doc.startTime.replace("T"," ")}`;
  }else{
    timeTip.innerText = "⚠️ 管理员未设置开始时间";
  }
  spinBtn.disabled = !canStart;

  initNickSelect();
  drawWheel();
  renderMyPrizes();
  renderTradeList();
  result.textContent = "";
  wantText.value = "";
}

function initNickSelect(){
  let html = '<option value="">-- 请选择昵称 --</option>';
  nickNameList.forEach(n=>html += `<option value="${n}">${n}</option>`);
  nickSelect.innerHTML = html;
}
function drawWheel(){
  const seg = 360/prizes.length;
  let html = "";
  prizes.forEach((p,i)=>{
    const img = p.img ? `<img src="${p.img}">` : "";
    html += `<div class="wheel-segment" style="background:${p.color};transform:rotate(${seg*i}deg)"><div class="prize-tag">${img}<span>${p.text}</span></div></div>`;
  });
  wheel.innerHTML = html;
}
function checkNick(){
  currentNick = nickSelect.value;
  if(!currentNick){nickTip.classList.remove("hidden");return false;}
  nickTip.classList.add("hidden");return true;
}

function saveCurrentData(){
  const doc = allDocs.find(d=>d.id===currentDocId);
  if(doc){
    doc.myPrizeList = [...myPrizeList];
    doc.tradeData = [...tradeData];
    doc.userAllPrize[currentNick] = [...myPrizeList];
  }
  localStorage.setItem(KEY, JSON.stringify(allDocs));
}

function startSpin(){
  if(!checkNick()||isSpinning)return;
  isSpinning = true; spinBtn.disabled = true; result.textContent = "";
  let total = prizes.reduce((a,b)=>a+b.prob,0);
  let r = Math.random()*total, idx=0;
  for(;idx<prizes.length;idx++){r-=prizes[idx].prob;if(r<=0)break;}
  const seg = 360/prizes.length;
  currentAngle += 360*5 + seg*idx + seg/2 - currentAngle%360;
  wheel.style.transform = `rotate(${currentAngle}deg)`;

  setTimeout(()=>{
    isSpinning = false;
    const got = prizes[idx].text;
    result.textContent = `🎉 ${currentNick} 抽到：${got}`;
    myPrizeList.push(got);
    selectedPrize = null;
    saveCurrentData();
    renderMyPrizes();
    const doc = allDocs.find(d=>d.id===currentDocId);
    spinBtn.disabled = !checkActivityTime(doc.startTime);
  },4000);
}

function renderMyPrizes(){
  if(myPrizeList.length===0){myPrizes.innerHTML='<span class="text-gray-400">暂无奖品</span>';return;}
  let html = "";
  myPrizeList.forEach((item,i)=>{
    const cls = selectedPrize===i ? "active ring-2 ring-blue-500 bg-blue-100" : "bg-gray-200";
    html += `<span class="prize-item ${cls} px-3 py-1 rounded" data-idx="${i}">${item}</span>`;
  });
  myPrizes.innerHTML = html;
  document.querySelectorAll(".prize-item").forEach(el=>{
    el.onclick = ()=>{
      const i = Number(el.dataset.idx);
      selectedPrize = selectedPrize===i ? null : i;
      renderMyPrizes();
    };
  });
}

function publishTradeOrder(){
  if(!checkNick()||selectedPrize===null)return alert("请先选中奖品");
  const want = wantText.value.trim();
  if(!want)return alert("请填写兑换需求");
  const give = myPrizeList[selectedPrize];
  tradeData.push({id:Date.now(), nick:currentNick, givePrize:give, wantPrize:want, time:new Date().toLocaleTimeString()});
  wantText.value = ""; selectedPrize = null;
  saveCurrentData(); renderMyPrizes(); renderTradeList();
}

function renderTradeList(){
  if(tradeData.length===0){tradeList.innerHTML="";emptyTip.classList.remove("hidden");return;}
  emptyTip.classList.add("hidden");
  let html = "";
  tradeData.forEach(t=>{
    html += `<div class="trade-item"><div class="text-sm mb-2">${t.time}｜${t.nick} 出让${t.givePrize}→求换${t.wantPrize}</div>
    <button class="doTrade bg-orange-500 text-white px-3 py-1 rounded text-sm" data-id="${t.id}" data-give="${t.givePrize}" data-want="${t.wantPrize}" data-from="${t.nick}">立即置换</button></div>`;
  });
  tradeList.innerHTML = html;
  document.querySelectorAll(".doTrade").forEach(btn=>{
    btn.onclick = ()=>handleTrade(btn);
  });
}

function handleTrade(btn){
  if(!checkNick())return;
  const id = Number(btn.dataset.id);
  const fromNick = btn.dataset.from;
  const givePrize = btn.dataset.give;
  const wantPrize = btn.dataset.want;

  const idx = tradeData.findIndex(t=>t.id===id);
  if(idx<0)return;
  const myIdx = myPrizeList.findIndex(p=>p===wantPrize);
  if(myIdx<0)return alert(`你没有${wantPrize}，无法置换`);

  myPrizeList[myIdx] = givePrize;
  tradeData.splice(idx,1);

  const doc = allDocs.find(d=>d.id===currentDocId);
  doc.tradeRecord.push({
    time: new Date().toLocaleTimeString(),
    from: fromNick,
    give: givePrize,
    to: currentNick,
    want: wantPrize
  });

  saveCurrentData(); renderMyPrizes(); renderTradeList();
  alert("恭喜你置换成功！");
}

loadFromLocal();
renderActivitySelect();

goStep2.onclick = ()=>{
  const aid = activitySelect.value;
  if(!aid)return alert("请先选择活动");
  loadActivity(aid);
  step1.classList.add("step-hide");
  step2.classList.remove("step-hide");
};

spinBtn.onclick = startSpin;
publishTrade.onclick = publishTradeOrder;
</script>
